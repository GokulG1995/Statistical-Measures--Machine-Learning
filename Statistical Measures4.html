# a)Mean and Standard Deviation Method
def remove_outliers_std(df, column):
    mean = df[column].mean()
    std = df[column].std()
    upper_bound = mean + 3 * std
    lower_bound = mean - 3 * std
    df_filtered = df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]
    return df_filtered

# b) Percentile Method
def remove_outliers_percentile(df, column):
    upper_limit = df[column].quantile(0.95)
    lower_limit = df[column].quantile(0.05)
    df_filtered = df[(df[column] >= lower_limit) & (df[column] <= upper_limit)]
    return df_filtered

# c) IQR Method
def remove_outliers_iqr(df, column):
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    upper_limit = Q3 + 1.5 * IQR
    lower_limit = Q1 - 1.5 * IQR
    df_filtered = df[(df[column] >= lower_limit) & (df[column] <= upper_limit)]
    return df_filtered
